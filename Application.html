from sklearn.metrics import accuracy_score

def check_fitting(model, X_train, y_train, X_test, y_test, model_name):

    train_pred = model.predict(X_train)
    train_acc = accuracy_score(y_train, train_pred)


    test_pred = model.predict(X_test)
    test_acc = accuracy_score(y_test, test_pred)


    gap = train_acc - test_acc

    print(f"--- {model_name} ---")
    print(f"Training Accuracy: {train_acc:.4f}")
    print(f"Testing Accuracy:  {test_acc:.4f}")
    print(f"Gap:               {gap:.4f}")

    if train_acc < 0.70 and test_acc < 0.70:
        print("Diagnosis: UNDERFITTING ")
    elif gap > 0.10:
        print("Diagnosis: OVERFITTING ")
    else:
        print("Diagnosis: GOOD FIT ")
    print("\n")


check_fitting(lr, X_train_res_scaled, y_train_res, X_test_scaled, y_test, "Logistic Regression")
check_fitting(dt, X_train, y_train, X_test, y_test, "Decision Tree")
check_fitting(rf, X_train, y_train, X_test, y_test, "Random Forest")
